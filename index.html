        """ Esercitazione: query parametriche su DB scolastico (solo lettura) """

import sqlite3
from typing import List, Tuple

def carica_voti_classe(db_path: str, classe_id: int) -> List[Tuple[str, float]]:
    # Connessione sicura: niente string formatting sulla query
    conn = sqlite3.connect(db_path)
    conn.row_factory = sqlite3.Row
    cur = conn.cursor()
    sql = """
        SELECT s.cognome, s.nome, m.voto
        FROM studenti AS s
        JOIN materie_voti AS m ON m.studente_id = s.id
        WHERE s.classe_id = ?
          AND m.anno_scolastico = ?
        ORDER BY s.cognome, s.nome;
    """
    cur.execute(sql, (classe_id, '2026/2027'))
    return [(f"{row['cognome']} {row['nome']}", float(row['voto'])) for row in cur.fetchall()]

if __name__ == '__main__':
    risultati = carica_voti_classe('./data/scuola.db', 3)
    for nominativo, voto in risultati:
        print(f"{nominativo}: {voto:.1f}")
